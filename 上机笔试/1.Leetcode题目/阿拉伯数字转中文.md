# 阿拉伯数字转中文

以"亿"、“万”为分界线进行划分，之后只用写四位数以内的转换即可。



```python
number = ['零', '一', '二', '三', '四', '五', '六', '七', '八', '九']
unit = ["","十","百","千"]


def thousand_to_chinese(num): ##四位数以内的转换，从左到右写
    num = str(num)
    ans = ""
    for i in range(len(num)):
        if i >= 1 and num[i] == "0" and num[i-1] == "0":
            continue
        if num[i]!="0":
            ans += number[int(num[i])]+unit[len(num)-i-1]
        elif i < len(num)-1:
            ans +="零"
    if len(ans) and ans[-1] == "零":
        ans = ans[:-1]
    return ans

def to_chinese(num):
    units = ["","万","亿"]
    num = str(num)
    curr = len(num)-1
    slices = []
    while curr >= 0:

        curr -= 4

    print(slices)
    res = ""
```

### 中文转阿拉伯数字

```python
def chinese2num(chinese):
    def fourdigit(chinese):
        number_dic = {"零":0,"一":1,"二":2,"三":3,"四":4,"五":5,"六":6,"七":7,"八":8,"九":9}
        unit_dic = {"十":10,"百":100,"千":1000}
        ans = 0
        for i in range(len(chinese)):
            if i < len(chinese)-1 and chinese[i] in number_dic and chinese[i+1] in unit_dic:
                ans += unit_dic[chinese[i+1]]*number_dic[chinese[i]]
                i += 1
            if i == len(chinese)-1 and chinese[i] in number_dic:
                ans += number_dic[chinese[i]]
        return ans
    ans = 0
    chinese = chinese.split("亿")
    if len(chinese) == 2:
        ans += 1000000000*fourdigit(chinese[0])
        chinese = chinese[1]
    chinese = chinese.split("万")
    if len(chinese) == 2:
        ans += 10000*fourdigit(chinese[0])
        chinese = chinese[1]
    print(chinese)
    ans += fourdigit(chinese)
    print(ans)
```
